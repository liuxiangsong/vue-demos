# div 居中显示

<style>
  .container {
    width: 800px;
    height: 800px;
    border: 1px solid;
  }
  .content {
    width: 200px;
    height: 200px;
    border: 1px solid;
  }
</style>

<div class="container">
  <div class="content"></div>
</div>
  • flex
.container {
  display: flex;
  justify-content: center;
  align-items: center;
}
  • transform
.container {
  position: relative;
}
.content {
  position: absolute;
  left: 50%;
  top: 50%;
  transform: translate(-50%, -50%);
  /* top: 50%; transform: translateY(-50%); //垂直居中 */
}
  • table-cell + vertical-align
.contaner {
  display: table-cell;
  vertical-align: middle;
}

# 图片居中

  }
</style>

</div>
  • line-height + vertical-align
.container {
  line-height: 800px;
  text-align: center;
}

  vertical-align: middle;
}
